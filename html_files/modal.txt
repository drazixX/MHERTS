//script

    // Renaming File

    $(document).ready(function() {
        // Event listener for the rename file button click
        $('.rename-file-button').click(function() {
            // Extract data attributes from the button
            var historyId = $(this).data('history-id');
            var familyRecId = $(this).data('family-rec-id');
            var folderName = $(this).data('folder-name');
            var fileId = $(this).data('file-id');

            // Populate form fields in the rename modal with data attributes
            $('#history_id1').val(historyId);
            $('#FamilyRec_Id1').val(familyRecId);
            $('#folder_name1').val(folderName);
            $('#file_id1').val(fileId);
        });

        // Event binding for rename button click inside the modal
        $('#confirmRename').click(function() {
            $('#renameForm').submit();
        });
    });
